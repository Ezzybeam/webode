# Webode Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build Webode — a Geode-style mod loader for the browser version of Geometry Dash at www.geometrydash.com, with a GD-styled mod panel accessible via a button in the main menu.

**Architecture:** `webode.js` (new Chrome Override) is the core — it provides the mod registry, event system, and scene bridge. `index.html` loads it before the game module and lists mod scripts. `index-game.js` has minimal patches at `killPlayer`, `hitGround`, `_restartLevel`, and jump to emit Webode events.

**Tech Stack:** Vanilla JS, Phaser 3 (already loaded by the game), Chrome DevTools Overrides

```js
// game/webode.js
(function () {
  'use strict';

  const _handlers = {};
  const _keybinds = {};
  const _mods = [];
  let _scene = null;

  const Webode = {
    version: '0.1.0',
    mods: _mods,

    // Called by index-game.js patch when scene create() runs
    _onSceneReady(scene) {
      _scene = scene;
      _loadLogoTexture(scene);
      _mods.forEach(m => m._sceneReady && m._sceneReady(scene));
    },

    // Mod registration
    registerMod(def) {
      const mod = Object.assign({ enabled: false }, def);
      const api = {
        get enabled() { return mod.enabled; },
        toggle() {
          mod.enabled = !mod.enabled;
          mod.enabled ? mod.onEnable && mod.onEnable() : mod.onDisable && mod.onDisable();
          Webode.ui.toast(mod.name + (mod.enabled ? ' ON' : ' OFF'), mod.enabled ? '#0c8' : '#c44');
          _refreshPanel();
        },
        addKeybind(key, fn) { Webode.addKeybind(key, fn); }
      };
      mod._api = api;
      _mods.push(mod);
      mod.onLoad && mod.onLoad(api);
    },

    // Event system
    on(event, fn) {
      (_handlers[event] = _handlers[event] || []).push(fn);
    },
    emit(event, data) {
      (_handlers[event] || []).forEach(fn => { try { fn(data); } catch(e) { console.warn('[Webode]', e); } });
    },

    // Keybind system
    addKeybind(key, fn) {
      (_keybinds[key.toLowerCase()] = _keybinds[key.toLowerCase()] || []).push(fn);
    },

    // Scene bridge
    get scene() { return _scene; },
    get player() { return _scene && _scene['_player']; },
    get state() { return _scene && _scene['_state']; },

    // UI namespace (filled in Task 5)
    ui: {
      toast(msg, color) { _showToast(msg, color); },
      openPanel() { _openPanel(); },
      closePanel() { _closePanel(); }
    }
  };

  // Keybind listener
  document.addEventListener('keydown', function (e) {
    if (e.ctrlKey || e.metaKey || e.altKey) return;
    const key = e.key.toLowerCase();
    (_keybinds[key] || []).forEach(fn => fn(e));
  });

  // Stubs — filled in later tasks
  function _loadLogoTexture(scene) {}
  function _showToast(msg, color) {}
  function _openPanel() {}
  function _closePanel() {}
  function _refreshPanel() {}

  window.Webode = Webode;
})();
```

```js
function _loadLogoTexture(scene) {
  if (scene.textures.exists('webode-logo')) {
    _onLogoReady(scene);
    return;
  }
  scene.textures.once('addtexture-webode-logo', () => _onLogoReady(scene));
  scene.textures.addBase64('webode-logo', WEBODE_LOGO);
}

function _onLogoReady(scene) {
  _buildMenuButton(scene);
}
```

```js
function _openPanel() {
  if (!_scene || _panelOpen) return;
  _panelOpen = true;

  const scene = _scene;
  const W = 280, H = 200;
  const cx = scene.scale.width / 2, cy = scene.scale.height / 2;

  _panelContainer = scene.add.container(cx, cy).setDepth(100).setScrollFactor(0);

  // Dark background
  const bg = scene.add.graphics();
  bg.fillStyle(0x000000, 0.85);
  bg.fillRoundedRect(-W/2, -H/2, W, H, 12);
  bg.lineStyle(2, 0xaa88ff, 1);
  bg.strokeRoundedRect(-W/2, -H/2, W, H, 12);
  _panelContainer.add(bg);

  // Title
  const title = scene.add.text(0, -H/2 + 18, 'Webode Mods', {
    fontFamily: 'Arial Black, Arial',
    fontSize: '14px',
    color: '#cc99ff',
    stroke: '#000',
    strokeThickness: 3
  }).setOrigin(0.5);
  _panelContainer.add(title);

  // Webode logo in title
  const logo = scene.add.image(-W/2 + 20, -H/2 + 18, 'webode-logo')
    .setDisplaySize(22, 22);
  _panelContainer.add(logo);

  // Close button (X)
  const closeBtn = scene.add.text(W/2 - 14, -H/2 + 8, '✕', {
    fontFamily: 'Arial',
    fontSize: '14px',
    color: '#ffffff'
  }).setOrigin(0.5).setInteractive({ useHandCursor: true });
  closeBtn.on('pointerdown', () => _closePanel());
  closeBtn.on('pointerover', () => closeBtn.setColor('#ff8888'));
  closeBtn.on('pointerout', () => closeBtn.setColor('#ffffff'));
  _panelContainer.add(closeBtn);

  // Divider
  const div = scene.add.graphics();
  div.lineStyle(1, 0x886699, 0.6);
  div.lineBetween(-W/2 + 10, -H/2 + 30, W/2 - 10, -H/2 + 30);
  _panelContainer.add(div);

  // Mod rows
  _buildModRows(scene, W, H);

  // Escape key to close
  scene.input.keyboard.once('keydown-ESC', () => _closePanel());
}
```

- [ ] **Step 3: Implement _buildModRows**

```js
function _buildModRows(scene, W, H) {
  const startY = -H/2 + 48;
  const rowH = 34;

  _mods.forEach((mod, i) => {
    const y = startY + i * rowH;

    // Mod name
    const nameText = scene.add.text(-W/2 + 14, y, mod.name, {
      fontFamily: 'Arial Black, Arial',
      fontSize: '11px',
      color: '#ffffff',
      stroke: '#000',
      strokeThickness: 2
    }).setOrigin(0, 0.5);
    _panelContainer.add(nameText);

    // Version
    const verText = scene.add.text(-W/2 + 14, y + 11, 'v' + mod.version + ' by ' + (mod.author || '?'), {
      fontFamily: 'Arial',
      fontSize: '8px',
      color: '#aaaaaa'
    }).setOrigin(0, 0.5);
    _panelContainer.add(verText);

    // Toggle button
    const toggleBg = scene.add.graphics();
    const btnX = W/2 - 40, btnY = y;
    const on = mod.enabled;
    _drawToggle(toggleBg, btnX, btnY, on);
    _panelContainer.add(toggleBg);

    const toggleHit = scene.add.rectangle(btnX, btnY, 44, 22, 0x000000, 0)
      .setInteractive({ useHandCursor: true });
    toggleHit.on('pointerdown', () => {
      mod._api.toggle();
    });
    _panelContainer.add(toggleHit);
  });
}

function _drawToggle(gfx, x, y, on) {
  gfx.clear();
  gfx.fillStyle(on ? 0x44cc88 : 0x884444, 1);
  gfx.fillRoundedRect(x - 22, y - 11, 44, 22, 11);
  gfx.fillStyle(0xffffff, 1);
  gfx.fillCircle(on ? x + 9 : x - 9, y, 8);
  gfx.lineStyle(1, on ? 0x22aa66 : 0x552222, 1);
  gfx.strokeRoundedRect(x - 22, y - 11, 44, 22, 11);
}
```

```js
function _showToast(msg, color) {
  // HTML overlay toast — works even before scene is ready
  let el = document.getElementById('webode-toast');
  if (!el) {
    el = document.createElement('div');
    el.id = 'webode-toast';
    el.style.cssText = [
      'position:fixed', 'top:14px', 'left:50%', 'transform:translateX(-50%)',
      'z-index:99999', 'padding:5px 14px', 'border-radius:6px',
      'font:bold 13px/1.4 Arial Black,Arial', 'pointer-events:none',
      'transition:opacity .3s', 'opacity:0', 'white-space:nowrap'
    ].join(';');
    document.body.appendChild(el);
  }
  el.style.background = color || '#446';
  el.style.color = '#fff';
  el.style.border = '1.5px solid rgba(255,255,255,0.25)';
  el.textContent = msg;
  el.style.opacity = '1';
  clearTimeout(el._t);
  el._t = setTimeout(() => { el.style.opacity = '0'; }, 1800);
}
```
